Is Single Value#

Imports#

from sklearn.datasets import load_iris
import pandas as pd
from deepchecks.tabular.checks.integrity.is_single_value import IsSingleValue

Load Data#

iris = load_iris()
X = iris.data
df = pd.DataFrame({'a':[3,4,1], 'b':[2,2,2], 'c':[None, None, None], 'd':['a', 4, 6]})
df
